#include <bits/stdc++.h> 
vector<int> nextPermutation(vector<int> &a, int n)
{
    //  Write your code here.
    int id1=0,id2=0,i;
    for(i=n-2;i>=0;i--){
        if(a[i]<a[i+1]){
            id1=i;
            break;
        }
    }
    for(int j=n-1;j>id1;j--){
        if(a[j]>a[id1]){
            id2=j;
            break;
        }
    }
    swap(a[id1],a[id2]);
    if(i>=0)
    reverse(a.begin()+id1+1,a.end());
    else{
      reverse(a.begin(),a.end());
    }
    return a;
}
